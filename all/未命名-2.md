digraph RBF_Network {
    rankdir=LR;          // 从左到右排版
    splines=line;        // 采用直线连接
    nodesep=0.7;         // 节点上下间距
    ranksep=2.5;         // 层级左右间距

    node [shape=circle, fontname="Times-Italic", fontsize=20, width=0.8, fixedsize=true];
    edge [arrowsize=0.8];

    // --- 输入层 ---
    subgraph cluster_0 {
        color=invis;
        x1 [label=<x<sup>(1)</sup>>];
        x2 [label=<x<sup>(2)</sup>>];
        xdots [shape=plaintext, label="⋮", fontname="Times-Roman", fontsize=26, width=0.1, height=0.5];
        xD [label=<x<sup>(D)</sup>>];
        label="输入层\n(D维决策变量)";
        fontname="SimHei"; fontsize=16;
    }

    // --- 隐含层 ---
    subgraph cluster_1 {
        color=invis;
        phi1 [label=<φ<sub>1</sub>>];
        phi2 [label=<φ<sub>2</sub>>];
        phidots [shape=plaintext, label="⋮", fontname="Times-Roman", fontsize=26, width=0.1, height=0.5];
        phiN [label=<φ<sub>N</sub>>];
        label="隐含层\n(N个径向基核函数)";
        fontname="SimHei"; fontsize=16;
    }

    // --- 输出层 ---
    subgraph cluster_2 {
        color=invis;
        Sigma [label=<<b>Σ</b>>, fontname="Times-Roman", fontsize=24];
        label="输出层\n(线性加权求和)";
        fontname="SimHei"; fontsize=16;
    }

    // 最终输出节点
    Output [shape=plaintext, label=<f(x)>, fontname="Times-Italic", fontsize=20, width=0.5];

    // 强制同层节点对齐
    {rank=same; x1; x2; xdots; xD;}
    {rank=same; phi1; phi2; phidots; phiN;}
    {rank=same; Sigma;}
    {rank=same; Output;}

    // 利用隐形边固定纵向顺序
    edge [style=invis];
    x1 -> x2 -> xdots -> xD;
    phi1 -> phi2 -> phidots -> phiN;

    // --- 连线：输入层到隐含层 ---
    edge [style=solid, color=gray60, arrowhead=normal];
    x1 -> {phi1, phi2, phiN};
    x2 -> {phi1, phi2, phiN};
    xD -> {phi1, phi2, phiN};

    // --- 连线：隐含层到输出层 ---
    edge [color=black, arrowhead=normal, fontname="Times-Italic", fontsize=18];
    phi1 -> Sigma [label=< w<sub>1</sub> >];
    phi2 -> Sigma [label=< w<sub>2</sub> >];
    phiN -> Sigma [label=< w<sub>N</sub> >];

    // 输出线
    edge [label=""];
    Sigma -> Output;
}
